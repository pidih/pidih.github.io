    UAT环境监控docker方式部署

    一、安装docker 环境

    pass

    二、下载prometheus、alertmanager 、pushgateway、dingtalk-webhook-pormetheus等镜像

    docker pull prom/prometheus
docker pull prom/alertmanager
docker pull prom/pushgateway
docker pull timonwong/prometheus-webhook-dingtalk

    三、添加配置文件启动docker

    1、prometheus

    配置文件：

    [root@wrp-uat-249 script]# vim /opt/monitor/prometheus/prometheus.yml
global:
  scrape_interval:     15s
  evaluation_interval: 15s 
alerting:
  alertmanagers:
  - static_configs:
    - targets:
      - 172.16.0.88:9093

rule_files:
   - "/data/alertrule/*.yaml"
scrape_configs:
  - job_name: 'prometheus'
    static_configs:
    - targets: ['172.16.0.88:9090']

  - job_name: 'Alertmanager'
    static_configs:
    - targets: ['172.16.0.88:9093']

  - job_name: "node_exporter"
    file_sd_configs:
      - files: ['/data/alertrule/nodegroup/node.json']

  - job_name: 'pushgateway'
    static_configs:
      - targets: ['172.16.0.88:9091']
        labels:
          instance: 172.16.0.88
          hostvm: pushgateway_249

  - job_name: "NacosUAT"
    metrics_path: /nacos/actuator/prometheus
    static_configs:
    - targets:
      - 172.16.0.94:8848
      labels:
        instance: 172.16.0.94
        nacosnode: uatnacos_49

  - job_name: 'bladeX-UAT'
    metrics_path: /actuator/prometheus
    scrape_interval: 5s
    consul_sd_configs:
      #必须保证prometheus能调用否则不会显示
      #若是docker部署必须保证网络与各服务调通
      - server: '172.16.0.92:7002' #填写实现consul-api的blade-admin服务ip:端口
        #匹配所有service
        services: []
    relabel_configs:
      #service 源标签
      - source_labels: [__meta_consul_service]
        #匹配 "blade" 开头的service
        regex: "blade*"
        #执行的动作
        action: drop
        #将service的label重写为application
      - source_labels: [__meta_consul_service]
        target_label: application
      - source_labels: [__meta_consul_service_address]
        target_label: instance

    启动prometheus容器

    docker run -d --name=prometheus \
        -p 9090:9090 \
        -v /opt/monitor/prometheus/prometheus.yml:/etc/prometheus/prometheus.yml \
        -v /opt/monitor/alertrule:/data/alertrule \
        prom/prometheus \
        --web.enable-lifecycle \
       --config.file=/etc/prometheus/prometheus.yml   \
       --storage.tsdb.path=/prometheus \
       --web.console.libraries=/usr/share/prometheus/console_libraries \
       --web.console.templates=/usr/share/prometheus/consoles

    告警规则

    bladeX-UAT blade服务存活告警

    [root@wrp-uat-249 script]# vim /opt/monitor/alertrule/bladex_alert.yaml
groups:
  - name: bladerule
    rules:
      - alert: BladeXapplications
        expr: up{job="bladeX-UAT"} == 0
        for: 2m
        labels:
          level: warning
          env: UAT
        annotations:
          summary: "Host {{ $labels.instance }} of {{ $labels.application }}: has been down"

    nacos告警 nacos存活告警、nacos配置文件变动、nacos服务注册在线数等

    [root@wrp-uat-249 script]# vim /opt/monitor/alertrule/nacosalert.yaml
groups:
- name: nacos_monitor
  rules:
  - alert: nacos_alive
    expr: up{job="NaocsUAT"} == 0
    for: 3m
    labels:
      severity: crirical
    annotations:
      description: " {{ $labels.instance }} of job of {{ $labels.job }} has been down for more than 5 minutes."
      summary: "{{ $labels.instance }} down,up=={{ $value }} // up=0表示监控已断开连接或监控点已下线，naocs已断开连接"

  - alert: nacos_configcount
    expr: nacos_monitor{job="NacosUAT",name="configCount"} < 12
    for: 5m
    labels:
      name: configCount
      severity: Warning
    annotations:
      summary: " {{ $labels.service }} nacos配置文件有变动"
      description: " nacos [{{ $labels.job }}] configCount={{ $value }},配置文件有新增或减少"

  - alert: nacos_servicecount
    expr: nacos_monitor{job="NacosUAT",name="serviceCount"} < 17
    for: 2m
    labels:
      name: serviceCount
      severity: crirical
    annotations:
      summary: " {{ $labels.job }} naocs 注册服务状态异常"
      description: " nacos [{{ $labels.name }}]  注册服务原本为17个。现有异常,目前在线服务数量 {{ $value }}"

    node_exproter告警 cpu memory disk node节点存活等告警

    [root@wrp-uat-249 script]# vim /opt/monitor/alertrule/node_alert.yaml
groups:
- name: UP
  rules:
  - alert: node
    expr: up{job="node_exporter"} == 0
    for: 1m
    labels:
      severity: crirical
    annotations:
      description: " {{ $labels.instance }} of job of {{ $labels.job }} has been down for more than 5 minutes."
      summary: "{{ $labels.instance }} down,up=={{ $value }} // up=0表示监控已断开连接，机器已断开连接"
  - alert: HostMemory Usage
    expr: (node_memory_MemTotal_bytes - (node_memory_MemFree_bytes + node_memory_Buffers_bytes + node_memory_Cached_bytes)) / node_memory_MemTotal_bytes * 100 >  85
    for: 1m
    labels:
      name: Memory
      severity: Warning
    annotations:
      summary: " {{ $labels.service}} 内存飙升 "
      description: "主机内存使用率超过80%."
      value: "{{ $value }}"
  - alert: HostCPU Usage
    expr: sum(avg without (cpu)(irate(node_cpu_seconds_total{mode!='idle'}[5m]))) by (instance,service) > 0.80
    for: 1m
    labels:
      name: CPU
      severity: Warning
    annotations:
      summary: " {{ $labels.service }} CPU飙升"
      description: "主机CPU使用率超过80%."
      value: "{{ $value }}"
  - alert: HostLoad 
    expr: node_load5 > 4
    for: 1m
    labels:
      name: Load
      severity: Warning
    annotations:
      summary: "{{ $labels.service }} 负荷飙升"
      description: " 主机负载5分钟超过4."
      value: "{{ $value }}"
  - alert: HostFilesystem Usage
    expr: 1-(node_filesystem_free_bytes / node_filesystem_size_bytes) >  0.8
    for: 1m
    labels:
      name: Disk
      severity: Warning
    annotations:
      summary: " {{ $labels.service }} 分区快满了"
      description: " 主机 [ {{ $labels.mountpoint }} ]分区使用超过80%."
      value: "{{ $value }}%"
  - alert: HostDiskio
    expr: irate(node_disk_writes_completed_total{job=~"Host"}[1m]) > 10
    for: 1m
    labels:
      name: Diskio
      severity: Warning
    annotations:
      summary: " {{ $labels.service }} IO读写过大"
      description: " 主机 [{{ $labels.device }}]磁盘1分钟平均写入IO负载较高."
      value: "{{ $value }}iops"
  - alert: Network_receive
    expr: irate(node_network_receive_bytes_total{device!~"lo|bond[0-9]|cbr[0-9]|veth.*|virbr.*|ovs-system"}[5m]) / 1048576  > 3 
    for: 1m
    labels:
      name: Network_receive
      severity: Warning
    annotations:
      summary: " {{ $labels.service }} 网络进来流量飙升"
      description: " 主机 [{{ $labels.device }}] 网卡5分钟平均接收流量超过3Mbps."
      value: "{{ $value }}3Mbps"
  - alert: Network_transmit
    expr: irate(node_network_transmit_bytes_total{device!~"lo|bond[0-9]|cbr[0-9]|veth.*|virbr.*|ovs-system"}[5m]) / 1048576  > 3
    for: 1m
    labels:
      name: Network_transmit
      severity: Warning
    annotations:
      summary: " {{ $labels.service }} 网络出去流量飙升"
      description: " 主机 [{{ $labels.device }}] 网卡5分钟内平均发送流量超过3Mbps."
      value: "{{ $value }}3Mbps"
