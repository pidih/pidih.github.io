    Elasticsearch7.8.1 和 kibana集群部署

    日志 elasticsearch;elk;es
    发布时间 : 2022-09-04 15:45
    阅读 :
    1. 环境说明
    2. 部署前准备
      1. 1、下载好安装包
      2. 2、服务器准备
      3. 3、elasticsearch部署
      4. 4、kibana部署

    环境说明

    1、单机部署三节点

    2、节点：
    master - 172.16.0.6:9200

    slave1 - 172.16.0.6:9201

    slave2 - 172.16.0.6:9202

    3、版本 7.8.1

    4、下载地址： https://elasticsearch.cn/download/ 都是下载的tar包安装

    比如filebeat : https://artifacts.elastic.co/downloads/beats/filebeat/filebeat-7.8.1-linux-x86.tar.gz

    部署前准备

    1、下载好安装包

    Elasticsearch\Filebeat\Kibana

    2、服务器准备

    [root@elasticsearch ~]# cd /opt
##改服务器主机名，关闭防火墙，关闭selinux
[root@elasticsearch opt]# hostnamectl set-hostname elasticsearch
[root@elasticsearch opt]# systemctl stop firewalld && systemctl disable firewalld
[root@elasticsearch opt]# sed -i 's#SELINUX=enforcing#SELINUX=disabled#g' /etc/selinux/config
## 把文件、软件、连接最大化
[root@elasticsearch opt]# echo "vm.max_map_count=262144" >> /etc/sysctl.conf
[root@elasticsearch opt]# vim /etc/security/limits.conf 
* soft nofile 100001
* hard nofile 100002
root soft nofile 100001
root hard nofile 100002
* soft nproc 65536
* hard nproc 65536
[root@elasticsearch opt]#  sysctl -p

    3、elasticsearch部署

    ## 将下载好的Efk传到服务器/opt/tools目录，解压缩。
[root@elasticsearch opt]# mkdir tools 
[root@elasticsearch opt]# ls -l tools
total 668552
-rw-r--r-- 1 root root 318334518 Aug 25 14:25 elasticsearch-7.8.1-linux-x86_64.tar.gz
-rw-r--r-- 1 root root  28736539 Aug 25 14:26 filebeat-7.8.1-linux-x86_64.tar.gz
-rw-r--r-- 1 root root 337517217 Aug 25 14:29 kibana-7.8.1-linux-x86_64.tar.gz
[root@elasticsearch opt]# cd tools && tar -xzf elasticsearch-7.8.1-linux-x86_64.tar.gz 
[root@elasticsearch tools]# mv elasticsearch-7.8.1 ../es-master && cd ../es-master
## 配置文件修改
[root@elasticsearch es-master]# cd config/
[root@elasticsearch es-master]# cp elasticsearch.yml elasticsearch.tmp
[root@elasticsearch es-master]# vim config/elasticsearch.yml
#集群名称，按照自己的需求调整
cluster.name: my-application
# 节点名称
node.name: master
# 设置data存储目录
path.data: /data/elasticsearch/es-master/data

# 设置logs日志的目录
path.logs: /data/elasticsearch/es-master/log

# 设置内存不使用交换分区
bootstrap.memory_lock: false

# 设置允许所有ip可以连接该elasticsearch，这里根据项目需求自行修改
network.host: 0.0.0.0

# 开启监听的端口，默认为9200
http.port: 9200
discovery.seed_hosts: ["127.0.0.1:9300","127.0.0.1:9301","127.0.0.1:9302"]
cluster.initial_master_nodes: ["master","node-1","node-2"]
[root@elasticsearch es-master]# mkdir -p /data/elasticsearch/es-master/data
[root@elasticsearch es-master]# mkdir -p /data/elasticsearch/es-master/log
[root@elasticsearch es-master]# mkdir -p /data/elasticsearch/es-slave1/{data,log}
[root@elasticsearch es-master]# mkdir -p /data/elasticsearch/es-slave2/{data,log}
## 复制两个节点 es-slave1 es-slave2
[root@elasticsearch es-master]# cd /opt && cp -rpa es-master es-slave1
[root@elasticsearch opt]# cp -rpa es-master es-slave2
[root@elasticsearch opt]# vim es-slave1/config/elasticsearch.yml
#集群名称，按照自己的需求调整
cluster.name: my-application
# 节点名称
node.name: node-1
# 设置data存储目录
path.data: /data/elasticsearch/es-slave1/data

# 设置logs日志的目录
path.logs: /data/elasticsearch/es-slave1/log

# 设置内存不使用交换分区
bootstrap.memory_lock: false

# 设置允许所有ip可以连接该elasticsearch，这里根据项目需求自行修改
network.host: 0.0.0.0

# 开启监听的端口，默认为9200
http.port: 9201
discovery.seed_hosts: ["127.0.0.1:9300","127.0.0.1:9301","127.0.0.1:9302"]
cluster.initial_master_nodes: ["master","node-1","node-2"]

[root@elasticsearch opt]# vim  es-slave2/config/elasticsearch.yml
#集群名称，按照自己的需求调整
cluster.name: my-application
# 节点名称
node.name: node-2
# 设置data存储目录
path.data: /data/elasticsearch/es-slave2/data

# 设置logs日志的目录
path.logs: /data/elasticsearch/es-slave2/log

# 设置内存不使用交换分区
bootstrap.memory_lock: false

# 设置允许所有ip可以连接该elasticsearch，这里根据项目需求自行修改
network.host: 0.0.0.0

# 开启监听的端口，默认为9202
http.port: 9202
discovery.seed_hosts: ["127.0.0.1:9300","127.0.0.1:9301","127.0.0.1:9302"]
cluster.initial_master_nodes: ["master","node-1","node-2"]

[root@elasticsearch opt]# groupadd esgroup && useradd esuser -g esgroup
[root@elasticsearch opt]# passwd esuser 
espassword
[root@elasticsearch opt]# chown -R esuser.esgroup es*
[root@elasticsearch opt]# chown -R esuser.esgroup /data/elasticsearch

    PS： discovery.seed_hosts 是用来连接选举的，默认是9300，当9300被占用，其他服务会自动在监听端口上加加100， 比如 监听9201，选举端口会开启9301端口。如果选举端口连接不上，服务连接不了选举将启动失败。所以尽可能把所有选举监控端口都填上。

    4、kibana部署

    [root@elasticsearch opt]# cd tools/ && tar -xzf kibana-7.8.1-linux-x86_64.tar.gz 
[root@elasticsearch tools]# mv kibana-7.8.1 ../kibana && chown -R esuser.esgroup ../kibana
