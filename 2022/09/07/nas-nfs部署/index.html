    部署NFS

    1、创建目录和安装 nfs

    [root@localhost ~]# mkdir /data
[root@localhost ~]# mount /dev/nas/acnas /data
[root@localhost ~]# df -lh
Filesystem               Size  Used Avail Use% Mounted on
devtmpfs                 3.9G     0  3.9G   0% /dev
tmpfs                    3.9G     0  3.9G   0% /dev/shm
tmpfs                    3.9G  8.9M  3.9G   1% /run
tmpfs                    3.9G     0  3.9G   0% /sys/fs/cgroup
/dev/mapper/centos-root   48G  2.1G   46G   5% /
/dev/sda2                197M  159M   39M  81% /boot
tmpfs                    783M     0  783M   0% /run/user/0
/dev/mapper/nas-acnas    4.0T   33M  4.0T   1% /data

vim /etc/exports
/data/nas 192.168.0.0/16(rw,root_squash,all_squash,sync,anonuid=1000,anongid=1000)
sudo exportfs -r
export fs -av
export -av
exportfs -av
