    2.主服务 prometheus server

    主服务器，负责收集和存储时间序列数据，负责实现对监控数据的获取，存储以及查询。

    端口：9090

    wget https://github.com/prometheus/prometheus/releases/download/v2.35.0-rc0/prometheus-2.35.0-rc0.linux-amd64.tar.gz
tar zxvf prometheus-2.35.0-rc0.linux-amd64.tar.gz 
mv prometheus-2.35.0-rc0.linux-amd64 prometheus
cd prometheus-2.35.0-rc0.linux-amd64
nohup ./prometheus --config.file=prometheus.yml > ./prometheus.log 2>&1 &

    可选指令

    --config.file= 指定配置文件目录
--storage.tsdb.path= 指定监控指标数据存储的路径
--storage.tsdb.retention=15d 历史数据最大保留时间，默认15day
--web.console.templates="/data/prometheus/consoles" 指定控制台模板目录路径
--web.console.libraries="/data/prometheus/console_librarics" 指定控制台库目录路径
--web.max-connections=512 设置最大连接数
--web.external-url
--web.listen-address="0.0.0.0:9090"  默认监控端口
--web.enable-lifecyle 开启热加载(1.kill -Hub pid 2. curl -X POST http://ip:port/-/reload)

    配置文件

    global #全局配置
  scrape_interval #每次数据采集的时间间隔，默认1s
  scrape_timeout  #采集请求超时时间，默认为10s
  evaluation_interval #执行rules的频率 默认1s
  external_labels #与外部系统通信时添加到任意时间序列或告警用的外部标签
scrape_configs #配置被采集数据节点
  job_name #全局唯一名称
    scrape_interval #默认等于global内的参数，设置后覆盖global的值
    scrape_timeout #默认等于global内的参数，设置后覆盖global的值
    metrics_path #从targets获取metric的http资源路径，默认为/metrics
    honor_labels #Prometheus如何处理标签之间的冲突，若设置为true，则通过保留标签来解决标签冲突进行数据值采集。若设置为false，则通过重名名解决标签冲突
    scheme #用于请求的协议方式，默认是http
    params #数据采集访问http url 设定的请求参数
    relabel_configs #采集数据重置标签配置
    metrics_relabel_configs #metric重置标签配置
    sample_limie #对样本(sample)数量的每次采集进行限制，超过限制数据为失败，默认为0，表示无限制

    3.node信息采集器 node_exporter

    监控样本数据的收集

    端口：9100

    wget https://github.com/prometheus/node_exporter/releases/download/v1.3.1/node_exporter-1.3.1.linux-amd64.tar.gz
tar
mv
cd
nohup ./node_exporter > node_exporter.log &
#在主服务Prometheus的配置文件里添加job
cat ../promethues/promethues.xml
scrape_configs:
  # The job name is added as a label `job=<job_name>` to any timeseries scraped from this config.
  - job_name: "prometheus"

    # metrics_path defaults to '/metrics'
    # scheme defaults to 'http'.

    static_configs:
      - targets: ["localhost:9090"]
  - job_name: "node_exporter"
    static_configs:
      - targets: ["localhost:9100"]

    4.可视化grafana

    grafana 是一款采用 Go 语言编写的开源应用，主要用于大规模指标数据的可视化展现，是网络架构和应用分析中最流行的时序数据展示工具，目前已经支持绝大部分常用的时序数据库

    端口：3000

    wget https://dl.grafana.com/enterprise/release/grafana-enterprise-8.4.6.linux-amd64.tar.gz
tar zxvf grafana-enterprise-8.4.6.linux-amd64.tar.gz 
mv grafana-8.4.6/ grafana
cd ../grafana
nohup ./bin/grafana-server > ./grafana.log 2>&1 &

    6.pushgateway

    端口：9091

    wget https://github.com/prometheus/pushgateway/releases/download/v1.4.2/pushgateway-1.4.2.linux-amd64.tar.gz
tar zxvf pushgateway-1.4.2.linux-amd64.tar.gz

cd pushgateway

    5.prometheus

    mkdir -pv /data/prometheus/config
cat > /data/prometheus/config/prometheus.yml << EOF
global:
  scrape_interval: 15s
  evaluation_interval: 15s
scrape_configs:
  - job_name: 'prometheus'
    static_configs:
    - targets: ['localhost:9090']
  - job_name: 'node-exporter'
    static_configs:
    - targets: ['192.168.20.10:9100']
  - job_name: 'cadvisor'
    static_configs:
    - targets: ['192.168.20.10:8080']
alerting:
  alertmanagers:
  - static_configs:
    - targets: ['192.168.20.10:9093']
rule_files:
  - "rules/*"    
EOF
mkdir -pv /data/prometheus/config/rules
cat > /data/prometheus/rules/test.yml << EOF
groups:
- name: CPU报警规则
  rules:
  - alert: CPU使用率告警
    expr: 100 - (avg by (instance)(irate(node_cpu_seconds_total{mode="idle"}[1m]) )) * 100 > 50
    for: 1m
    labels:
      severity: warning
    annotations:
      summary: "CPU使用率正在飙升。"
      description: "CPU使用率超过50%（当前值：{{ $value }}%）"
- name: 磁盘使用率报警规则
  rules:
  - alert: 磁盘使用率告警
    expr: 100 - node_filesystem_free_bytes{fstype=~"xfs|ext4"} / node_filesystem_size_bytes{fstype=~"xfs|ext4"} * 100 > 80
    for: 20m
    labels:
      severity: warning
    annotations:
      summary: "硬盘分区使用率过高"
      description: "分区使用大于80%（当前值：{{ $value }}%）"
- name: 内存报警规则
  rules:
  - alert: 内存使用率告警
    expr: (1 - (node_memory_MemAvailable_bytes / (node_memory_MemTotal_bytes))) * 100 > 50
    for: 1m
    labels:
      severity: warning
    annotations:
      summary: "服务器可用内存不足。"
      description: "内存使用率已超过50%（当前值：{{ $value }}%）"
- name: 实例存活告警规则
  rules:
  - alert: 实例存活告警
    expr: up == 0
    for: 10s
    labels:
      user: prometheus
      severity: warning
    annotations:
      summary: "主机宕机 !!!"
      description: "该实例主机已经宕机超过一分钟了。"
EOF

docker run  -d -p 9090:9090 \
-v /data/prometheus/config:/etc/prometheus/ \
-v /etc/localtime:/etc/localtime \
--name=prometheus \
--restart=always \
prom/prometheus \
--config.file=/etc/prometheus/prometheus.yml \
--web.enable-lifecycle  #开启热加载  culr -X POST:192.168.20.10:9090/-/reload
