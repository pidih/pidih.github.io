    前言

    本文主要介绍jetlinks自动化部署的流程，主要分为以下几步：

    1. 准备环境
    2. 部署jetlinks
    3. 部署插件
    4. 部署用户
    5. 部署设备
    6. 部署产品
    7. 部署规则

    准备环境

    1. 安装docker

    sudo apt-get install docker.io
sudo systemctl start docker
sudo systemctl enable docker

    2. 安装docker-compose

    sudo apt-get install docker-compose

    3. 安装mysql

    sudo apt-get install mysql-server           # 安装mysql
sudo mysql_secure_installation            # 安全设置
sudo systemctl start mysql                  # 启动mysql
sudo systemctl enable mysql                 # 开机启动mysql

    4. 安装redis

    sudo apt-get install redis-server            # 安装redis
sudo systemctl start redis                  # 启动redis
sudo systemctl enable redis                 # 开机启动redis

    5. 安装jdk

    sudo apt-get install openjdk-8-jdk

    6. 安装maven

    sudo apt-get install maven

    7. 安装nodejs

    sudo apt-get install nodejs

    8. 安装git

    sudo apt-get install git

    部署jetlinks

    1. 下载jetlinks源码

    git clone https://github.com/jetlinks/jetlinks-community.git

    2. 编译源码

    cd jetlinks-community
mvn clean install -DskipTests

    3. 启动jetlinks

    cd jetlinks-community/jetlinks-manager/target
java -jar jetlinks-manager.jar

    4. 访问管理后台

    访问http://localhost:9000

    默认账号：admin
    默认密码：123456

    5. 停止jetlinks

    cd jetlinks-community/jetlinks-manager/target
java -jar jetlinks-manager.jar --stop

    部署插件

    1. 下载插件源码

    git clone https://github.com/jetlinks/jetlinks-community.git

    2. 编译插件

    cd jetlinks-community/jetlinks-plugin
mvn clean install -DskipTests

    3. 启动插件

    cd jetlinks-community/jetlinks-plugin/target
java -jar jetlinks-plugin.jar

    4. 停止插件

    cd jetlinks-community/jetlinks-plugin/target
java -jar jetlinks-plugin.jar --stop

    jetlinks的jenkins自动化

    1. 修改jetlinks的原生pom文件

    jetlins需要映射的端口
    ports:
    - “8838:8848” # API端口
    - “1883-1890:1883-1890” # 预留
    - “8800-8810:8800-8810” # 预留
    - “5060-5061:5060-5061” # 预留

    另外我们需要将pom文件改成直接maven后可以推送到harbor仓库的方式

    1、在pom文件中的properties模块添加以下内容

        <properties>
        <!-- 推荐使用Harbor -->
        <docker.registry.url>registry.cn-shenzhen.aliyuncs.com</docker.registry.url>
        <docker.registry.host>http://${docker.registry.url}:2375</docker.registry.host>
        <docker.username>ancangshuju</docker.username>
        <docker.password>Docker365acd.com</docker.password>
        <docker.namespace>aciot-blade</docker.namespace>
        <docker.plugin.version>1.4.13</docker.plugin.version>
    </properties>
```xml
#### 2、在build模块中添加以下内容
```xml
<build>
    <plugins>
        <plugin>
            <groupId>org.apache.maven.plugins</groupId>
            <artifactId>maven-deploy-plugin</artifactId>
            <version>2.8.2</version>
            <configuration>
                <skip>true</skip>
            </configuration>
        </plugin>
        <plugin>
            <groupId>org.apache.maven.plugins</groupId>

            <artifactId>maven-compiler-plugin</artifactId>
            <version>3.8.0</version>
            <configuration>
                <source>1.8</source>
                <target>1.8</target>
            </configuration>
        </plugin>
    </plugins>
</build>

    3、在pluginManagement模块中添加以下内容

    <build>
    </pluginManagement>
                    <plugin>
                    <groupId>com.spotify</groupId>
                    <artifactId>dockerfile-maven-plugin</artifactId>
                    <version>${docker.plugin.version}</version>
                    <configuration>
                        <username>${docker.username}</username>
                        <password>${docker.password}</password>
                        <repository>${docker.registry.url}/${docker.namespace}/${project.artifactId}</repository>
                        <tag>${project.version}</tag>
                        <useMavenSettingsForAuth>true</useMavenSettingsForAuth>
                        <buildArgs>
                            <JAR_FILE>target/${project.build.finalName}.jar</JAR_FILE>
                        </buildArgs>
                    </configuration>
                    <!--子服务添加如下配置，运行 mvn deploy 命令便会自动打包镜像-->
                    <!--<executions>
                        <execution>
                            <id>default</id>
                            <goals>
                                <goal>build</goal>
                                <goal>push</goal>
                            </goals>
                        </execution>
                    </executions>-->
                </plugin>
        </pluginManagement>
</build>

    2. 在jenkins上创建流水线项目

    def servicesPort = ["blade-gateway":88,"blade-auth":8100,"blade-system":8106,"blade-resource":8010,"blade-swagger":18000,"blade-admin":7002,"blade-desk":8105,"blade-log":8103]
def selectProjectName = ["blade-auth","blade-gateway","blade-ops/blade-admin","blade-ops/blade-resource","blade-ops/blade-swagger","blade-service/blade-desk","blade-service/blade-system","blade-ops/blade-log"]
def AllProjectName = ["blade-gateway","blade-auth","blade-system","blade-desk","blade-admin","blade-swagger","blade-resource","blade-log"]
pipeline {
    agent any
    environment {
        git_url="http://git.365acd.com/newiotgroup/jetlinks-community.git"
        scannerHome="/opt/sonar-scanner4.7"
        scan_project="jetlins"
        app_name="jetlinks"
        _tags=sh(script: "echo `date '+%s'`", returnStdout: true).trim()
        
    }
    parameters {
        gitParameter (branch:'', branchFilter: 'origin/(.*)', defaultValue: 'master', description: '选择将要构建的分支', name: 'BRANCH', quickFilterEnabled: true, selectedValue: 'TOP', sortMode: 'DESCENDING_SMART', tagFilter: '*', type: 'PT_BRANCH_TAG', useRepository: env.GIT_URL)
    }
    
    stages {
        stage('clean') {
            steps {
            echo "-------------------------------------start clean-------------------------------------"   
            cleanWs(
                cleanWhenAborted: true, 
                cleanWhenFailure: true, 
                cleanWhenNotBuilt: true, 
                cleanWhenSuccess: true, 
                cleanWhenUnstable: true, 
                cleanupMatrixParent: true, 
                disableDeferredWipeout: true, //# 这个选项是关闭延时删除，立即删除。
                deleteDirs: true
                )
             echo "-------------------------------------successful clean-------------------------------------"   
            }
        }
        stage('git') {
            steps {
                echo "-------------------------------------start git  -------------------------------------" 
                checkout([$class: 'GitSCM', branches: [[name: "$BRANCH"]],  userRemoteConfigs: [[credentialsId: 'jenkinsgit', url: "${git_url}"]]])
                echo "-----------------------------------checkout successful ****-------------------------------------" 
            }
        }
        
        stage('docker build') {
            steps {
                echo "-------------------------------------start maven-------------------------------------"
                
                sh "find ./ -name Dockerfile  -type f|xargs  sed -i  's/test/dev/g' "
                sh "mvn install -Dmaven.test.skip=true"
                script {
                for(int i=0; i<selectProjectName.size(); i++){
                  def currentProjectName = selectProjectName[i];
                  sh "/data/script/change_tags.sh ${_tags} ${currentProjectName}";
                  sh "mvn -f ${currentProjectName}/pom.xml dockerfile:build dockerfile:push";
                  echo "${currentProjectName}完成镜像制作"
                }
                }
                echo "-------------------------------------maven build successful-------------------------------------"
            }
        }
        
        stage('deploy to server'){
            steps{
                script{
                    echo " -------------------------------------start upload-------------------------------------"
                    //清空deploy下的${app_name}.jar 
                    for(int i=0;i<AllProjectName.size();i++){
                      def currentProjectName = AllProjectName[i];
                      def port=servicesPort[currentProjectName];
                      sh """/usr/bin/ansible dev_aciot -m shell -a "docker pull 192.168.2.208/aciot-blade/${currentProjectName}:${_tags}" """ 
                      sh """/usr/bin/ansible dev_aciot -m shell -a "docker stop ${currentProjectName} && docker rm ${currentProjectName} || echo 'skip '" """ ;
                      sh """/usr/bin/ansible dev_aciot -m shell -a "docker run --name ${currentProjectName} -p ${port}:${port} -d 192.168.2.208/aciot-blade/${currentProjectName}:${_tags}" """ 
                    }
                    
                    echo "-------------------------------------upload successful-------------------------------------"
                 
                }
            }
        }
    }
    
    post {
        failure {
            wrap([$class: 'BuildUser']) {
                dingtalk(
                    robot: 'zhuyuanzhang',
                    type: 'MARKDOWN',
                    text: ["### 项目信息",
                        '---',
                        "- 项目主体: CheckCheck",
                        "- 子项目: 斗地主",
                        "![ddz](https://img95.699pic.com/xsj/06/os/zs.jpg)",
                        "",
                        '---',
                        "### 构建信息",
                        '---',
                        "- 构建编号: <font color=#dd3238>${BUILD_ID}</font>",
                        "- 构建状态: **<font color=#dd3238>${currentBuild.result}</font>**",
                        "- 项目地址: ${BUILD_URL}",
                        "- 构建日志: ${BUILD_URL}console",
                        "",
                        '---',
                        "### 执行人",
                        '---',
                        "- ${env.BUILD_USER}",
                        "",
                        '---',
                        "### 持续时间",
                        '---',
                        "- ${currentBuild.durationString}",
                        "",
                        '---']
                )
            }
        }
    }
    
}
